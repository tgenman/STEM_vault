$$
\sum_{i=1}^{n} x_i y_i \leq \left( \sum_{i=1}^{n} |x_i|^p \right)^{1/p} \left( \sum_{i=1}^{n} |y_i|^q \right)^{1/q}, \quad \text{where} \quad \frac{1}{p} + \frac{1}{q} = 1.
$$
