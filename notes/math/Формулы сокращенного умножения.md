$$(𝑎+𝑏)^2 =𝑎^2 +2𝑎𝑏+𝑏^2,$$
$$(𝑎−𝑏)^2 =𝑎^2 −2𝑎𝑏+𝑏^2,  $$
$$𝑎^2 −𝑏^2 = (𝑎+𝑏)(𝑎−𝑏),  $$
$$(𝑎+𝑏)^3 =𝑎^3 +3𝑎^2𝑏+3𝑎𝑏^2 +𝑏^3,$$
$$(𝑎−𝑏)^3 =𝑎^3 −3𝑎^2𝑏+3𝑎𝑏^2 −𝑏^3,$$
$$𝑎^3 +𝑏^3 = (𝑎+𝑏)(𝑎^2 −𝑎𝑏+𝑏^2),$$
$$𝑎^3 −𝑏^3 = (𝑎−𝑏)(𝑎^2 +𝑎𝑏+𝑏^2).$$
